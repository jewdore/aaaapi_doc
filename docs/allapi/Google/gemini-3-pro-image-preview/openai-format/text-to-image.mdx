```python Python
import requests
import json

url = "https://gptproto.com/v1/images/generations"
headers = {
    "Authorization": "sk-***********",
    "Content-Type": "application/json"
}

data = {
    "model": "gemini-3-pro-image-preview",
    "prompt": "A cute puppy playing in the park",
    "n": 1,
    "size": "1024x1024"
}

response = requests.post(url, headers=headers, json=data)
result = response.json()
print(json.dumps(result, indent=2))
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1/images/generations"

    payload := map[string]interface{}{
        "model": "gemini-3-pro-image-preview",
        "prompt": "A cute puppy playing in the park",
        "n": 1,
        "size": "1024x1024"
    }

    jsonData, _ := json.Marshal(payload)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "sk-***********")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```

## Parameters

| Parameter | Type | Required | Default | Description |
|-----------|------|----------|---------|-------------|
| `model` | string | ✅ Yes | - | The model to use for generating images (e.g., "gemini-3-pro-image-preview"). |
| `prompt` | string | ✅ Yes | - | The text prompt to generate images from. |
| `n` | integer | ❌ No | 1 | The number of images to generate. |
| `size` | string | ❌ No | 1024x1024 | The size of the generated images (e.g., "256x256", "512x512", "1024x1024"). |
| `quality` | string | ❌ No | standard | The quality of the generated images ("standard" or "hd"). |
| `response_format` | string | ❌ No | url | The format of the response ("url" or "b64_json"). |
| `style` | string | ❌ No | vivid | The style of the generated images ("vivid" or "natural"). |
