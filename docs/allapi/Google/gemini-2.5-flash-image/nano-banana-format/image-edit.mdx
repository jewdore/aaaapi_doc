```bash cURL
curl --location --request POST 'https://gptproto.com/api/v3/google/nano-banana-pro/image-edit' \
--header 'Authorization: GPTPROTO_API_KEY' \
--header 'Content-Type: application/json' \
--data-raw '{
    "prompt": "Put a hat on the cat",
    "images": [
        "https://tos.gptproto.com/resource/cat.png"

    "size": "1K",
    "aspect_ratio": "1:1",
    "enable_base64_output": false,
    "enable_sync_mode": false
}'
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/google/nano-banana-pro/image-edit"
    
    payload := map[string]interface{}{
        "prompt": "Put a hat on the cat",
        "images": []string{
            "https://tos.gptproto.com/resource/cat.png",
        },
        "output_format": "jpeg",
        "size": "1K",
        "aspect_ratio": "1:1",
        "enable_base64_output": false,
        "enable_sync_mode": false,
    }
    
    jsonData, _ := json.Marshal(payload)
    
    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")
    
    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        fmt.Println(err)
        return
    }
    defer resp.Body.Close()
    
    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    fmt.Println(result)
}
```
