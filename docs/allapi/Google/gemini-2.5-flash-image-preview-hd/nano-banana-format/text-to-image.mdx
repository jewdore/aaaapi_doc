## Initiate Request

```bash cURL
curl --location --request POST 'https://gptproto.com/api/v3/google/nano-banana-pro/text-to-image' \
--header 'Authorization: GPTPROTO_API_KEY' \
--header 'Content-Type: application/json' \
--data-raw '{
    "prompt": "Put a hat on the cat",
    "output_format": "jpeg",
    "size": "1K",
    "aspect_ratio": "1:1",
    "enable_base64_output": false,
    "enable_sync_mode": false
}'
```

```python Python
import requests
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/google/nano-banana-pro/text-to-image"

    payload := map[string]interface{}{
        "prompt":               "Put a hat on the cat",
        "output_format":        "jpeg",
        "size":                 "1K",
        "aspect_ratio":         "1:1",
        "enable_base64_output": false,
        "enable_sync_mode":     false,
    }

    jsonData, err := json.Marshal(payload)
    if err != nil {
        panic(err)
    }

    req, err := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    if err != nil {
        panic(err)
    }

    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, err := ioutil.ReadAll(resp.Body)
    if err != nil {
        panic(err)
    }

    fmt.Println(string(body))
}
```
