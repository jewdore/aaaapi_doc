## Initiate Request
<CodeGroup>

```bash cURL
curl --location --request POST 'https://gptproto.com/v1/images/generations' \
--header 'Authorization: GPTPROTO_API_KEY' \
--header 'Content-Type: application/json' \
--data-raw '{
    "model": "gemini-2.5-flash-image-preview-hd",
    "prompt": "a cat",
    "size":"1:1"
}'
```

```python Python
from openai import OpenAI

client = OpenAI(
    api_key="GPTPROTO_API_KEY",
    base_url="https://gptproto.com/v1"
)

response = client.images.generate(
    model="gemini-2.5-flash-image-preview-hd",
    prompt="a cat",
    size="1:1"
)

print(response)
```

```javascript JavaScript
import OpenAI from "openai";

const client = new OpenAI({
    apiKey: "GPTPROTO_API_KEY",
    baseURL: "https://gptproto.com/v1"
});

async function generateImage() {
    const response = await client.images.generate({
        model: "gemini-2.5-flash-image-preview-hd",
        prompt: "a cat",
        size: "1:1"
    });
    
    console.log(response);
}

generateImage();
```

```go Go
package main

import (
    "context"
    "fmt"
    "github.com/sashabaranov/go-openai"
)

func main() {
    config := openai.DefaultConfig("GPTPROTO_API_KEY")
    config.BaseURL = "https://gptproto.com/v1"
    client := openai.NewClientWithConfig(config)

    req := openai.ImageRequest{
        Model:  "gemini-2.5-flash-image-preview-hd",
        Prompt: "a cat",
        Size:   "1:1",
    }

    resp, err := client.CreateImage(context.Background(), req)
    if err != nil {
        fmt.Printf("Image generation error: %v\n", err)
        return
    }

    fmt.Println(resp)
}
```
