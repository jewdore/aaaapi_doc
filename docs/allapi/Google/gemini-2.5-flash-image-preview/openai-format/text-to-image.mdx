## Initiate Request
<CodeGroup>

```bash cURL
curl --location --request POST 'https://gptproto.com/v1/images/generations' \
--header 'Authorization: GPTPROTO_API_KEY' \
--header 'Content-Type: application/json' \
--data-raw '{
    "model": "gemini-2.5-flash-image-preview",
    "prompt": "a cat",
    "size":"1:1"
}'
```

```python Python
import requests
import json

url = "https://gptproto.com/v1/images/generations"

headers = {
    "Authorization": "GPTPROTO_API_KEY",
    "Content-Type": "application/json"
}

payload = {
    "model": "gemini-2.5-flash-image-preview",
    "prompt": "a cat",
    "size": "1:1"
}

response = requests.post(url, headers=headers, json=payload)
print(response.json())
```

```javascript JavaScript
const url = 'https://gptproto.com/v1/images/generations';

const headers = {
    'Authorization': 'GPTPROTO_API_KEY',
    'Content-Type': 'application/json'
};

const data = {
    model: 'gemini-2.5-flash-image-preview',
    prompt: 'a cat',
    size: '1:1'
};

fetch(url, {
    method: 'POST',
    headers: headers,
    body: JSON.stringify(data)
})
.then(response => response.json())
.then(data => console.log(data))
.catch(error => console.error('Error:', error));
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1/images/generations"
    
    payload := map[string]interface{}{
        "model":  "gemini-2.5-flash-image-preview",
        "prompt": "a cat",
        "size":   "1:1",
    }
    
    jsonData, _ := json.Marshal(payload)
    
    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")
    
    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()
    
    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
