## Initiate Request
<CodeGroup>

```bash cURL
curl --location --request POST 'https://gptproto.com/v1/chat/completions' \
--header 'Authorization: GPTPROTO_API_KEY' \
--header 'Content-Type: application/json' \
--data-raw '{
  "model": "gpt-4.1-2025-04-14",
  "messages": [
    {
      "role": "user",
      "content": "Who are you?"
    }
  ],
  "stream": false
}'
```

```python Python
import requests

url = "https://gptproto.com/v1/chat/completions"
headers = {
    "Authorization": "GPTPROTO_API_KEY",
    "Content-Type": "application/json"
}
data = {
    "model": "gpt-4.1-2025-04-14",
    "messages": [
        {
            "role": "user",
            "content": "Who are you?"
        }
    ],
    "stream": False
}

response = requests.post(url, headers=headers, json=data)
print(response.json())
```

```javascript JavaScript
const url = 'https://gptproto.com/v1/chat/completions';
const headers = {
    'Authorization': 'GPTPROTO_API_KEY',

};
const data = {
    model: 'gpt-4.1-2025-04-14',
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1/chat/completions"
    
    payload := map[string]interface{}{
        "model": "gpt-4.1-2025-04-14",
        "messages": []map[string]string{
            {
                "role":    "user",
                "content": "Who are you?",
            },
        },
        "stream": false,
    }
    
    jsonData, _ := json.Marshal(payload)
    
    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")
    
    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        fmt.Println("Error:", err)
        return
    }
    defer resp.Body.Close()
    
    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
