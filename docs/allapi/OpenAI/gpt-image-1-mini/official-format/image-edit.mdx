## Initiate Request
<CodeGroup>

```bash cURL
curl --location --request POST "https://gptproto.com/v1/images/edits" \
--header "Authorization: GPTPROTO_API_KEY" \
--form "model=gpt-image-1-mini" \
--form "image[]=@/cat.jpg" \
--form "prompt=Put a hat on the cat"
```

```python Python
import requests

url = "https://gptproto.com/v1/images/edits"
headers = {
    "Authorization": "GPTPROTO_API_KEY"
}

files = {
    "image[]": open("/cat.jpg", "rb")
}

data = {
    "model": "gpt-image-1-mini",
    "prompt": "Put a hat on the cat"
}

response = requests.post(url, headers=headers, files=files, data=data)
print(response.json())
```

```javascript JavaScript
const FormData = require('form-data');
const fs = require('fs');
const axios = require('axios');

const form = new FormData();
form.append('model', 'gpt-image-1-mini');
form.append('image[]', fs.createReadStream('/cat.jpg'));
form.append('prompt', 'Put a hat on the cat');

axios.post('https://gptproto.com/v1/images/edits', form, {
  headers: {
    'Authorization': 'GPTPROTO_API_KEY',
    ...form.getHeaders()
  }
})
.then(response => console.log(response.data))
.catch(error => console.error(error));
```

```go Go
package main

import (
    "bytes"
    "fmt"
    "io"
    "mime/multipart"
    "net/http"
    "os"
)

func main() {
    url := "https://gptproto.com/v1/images/edits"
    
    body := &bytes.Buffer{}
    writer := multipart.NewWriter(body)
    
    writer.WriteField("model", "gpt-image-1-mini")
    writer.WriteField("prompt", "Put a hat on the cat")
    
    file, _ := os.Open("/cat.jpg")
    defer file.Close()
    
    part, _ := writer.CreateFormFile("image[]", "cat.jpg")
    io.Copy(part, file)
    writer.Close()
    
    req, _ := http.NewRequest("POST", url, body)
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", writer.FormDataContentType())
    
    client := &http.Client{}
    resp, _ := client.Do(req)
    defer resp.Body.Close()
    
    respBody, _ := io.ReadAll(resp.Body)
    fmt.Println(string(respBody))
}
```
