## Initiate Request
<CodeGroup>

```bash cURL
curl --location --request POST 'https://gptproto.com/api/v3/openai/sora-2/image-to-video' \
--header 'Authorization: GPTPROTO_API_KEY' \
--header 'Content-Type: application/json' \
--data-raw '{
  "prompt": "Put a hat on the cat",
  "image": "https://tos.gptproto.com/resource/cat.png",
  "duration": 4
}'
```

```python Python
import requests
import json

url = "https://gptproto.com/api/v3/openai/sora-2/image-to-video"

headers = {
    "Authorization": "GPTPROTO_API_KEY",
    "Content-Type": "application/json"
}

payload = {
    "prompt": "Put a hat on the cat",
    "image": "https://tos.gptproto.com/resource/cat.png",
    "duration": 4
}

response = requests.post(url, headers=headers, data=json.dumps(payload))
print(response.json())
```

```javascript JavaScript
const url = 'https://gptproto.com/api/v3/openai/sora-2/image-to-video';

const headers = {
    'Authorization': 'GPTPROTO_API_KEY',
    'Content-Type': 'application/json'
};

const payload = {
    prompt: 'Put a hat on the cat',
    image: 'https://tos.gptproto.com/resource/cat.png',
    duration: 4
};

fetch(url, {
    method: 'POST',
    headers: headers,
    body: JSON.stringify(payload)
})
.then(response => response.json())
.then(data => console.log(data))
.catch(error => console.error('Error:', error));
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/openai/sora-2/image-to-video"

    payload := map[string]interface{}{
        "prompt":   "Put a hat on the cat",
        "image":    "https://tos.gptproto.com/resource/cat.png",
        "duration": 4,
    }

    jsonPayload, _ := json.Marshal(payload)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonPayload))
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
