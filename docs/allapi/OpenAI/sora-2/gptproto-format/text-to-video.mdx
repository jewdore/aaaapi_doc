## Initiate Request
<CodeGroup>

```bash cURL
curl --location --request POST 'https://gptproto.com/api/v3/openai/sora-2/text-to-video' \
--header 'Authorization: GPTPROTO_API_KEY' \
--header 'Content-Type: application/json' \
--data-raw '{
  "prompt": "A video of a horse dancing.",
  "size": "720*1280",
  "duration": 4
}'
```

```python Python
import requests
import json

url = "https://gptproto.com/api/v3/openai/sora-2/text-to-video"

headers = {
    "Authorization": "GPTPROTO_API_KEY",
    "Content-Type": "application/json"
}

payload = {
    "prompt": "A video of a horse dancing.",
    "size": "720*1280",
    "duration": 4
}

response = requests.post(url, headers=headers, json=payload)
print(response.json())
```

```javascript JavaScript
const axios = require('axios');

const url = 'https://gptproto.com/api/v3/openai/sora-2/text-to-video';

const headers = {
    'Authorization': 'GPTPROTO_API_KEY',
    'Content-Type': 'application/json'
};

const data = {
    prompt: 'A video of a horse dancing.',
    size: '720*1280',
    duration: 4
};

axios.post(url, data, { headers })
    .then(response => console.log(response.data))
    .catch(error => console.error(error));
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/openai/sora-2/text-to-video"

    payload := map[string]interface{}{
        "prompt":   "A video of a horse dancing.",
        "size":     "720*1280",
        "duration": 4,
    }

    jsonData, _ := json.Marshal(payload)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
