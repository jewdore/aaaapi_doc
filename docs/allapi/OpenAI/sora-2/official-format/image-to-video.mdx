## Initiate Request
<CodeGroup>

```bash cURL
curl --location --request POST 'https://gptproto.com/v1/videos' \
--header 'Authorization: GPTPROTO_API_KEY' \
--form 'model=sora-2' \
--form 'prompt=Put a hat on the cat' \
--form 'input_reference=@cat.jpg'
```

```python Python
import requests

url = "https://gptproto.com/v1/videos"
headers = {
    "Authorization": "GPTPROTO_API_KEY"
}
files = {
    "input_reference": open("cat.jpg", "rb")
}
data = {
    "model": "sora-2",
    "prompt": "Put a hat on the cat"
}

response = requests.post(url, headers=headers, files=files, data=data)
print(response.json())
```

```go Go
package main

import (
    "bytes"
    "io"
    "mime/multipart"
    "net/http"
    "os"
)

func main() {
    url := "https://gptproto.com/v1/videos"
    
    body := &bytes.Buffer{}
    writer := multipart.NewWriter(body)
    
    writer.WriteField("model", "sora-2")
    writer.WriteField("prompt", "Put a hat on the cat")
    
    file, _ := os.Open("cat.jpg")
    defer file.Close()
    part, _ := writer.CreateFormFile("input_reference", "cat.jpg")
    io.Copy(part, file)
    
    writer.Close()
    
    req, _ := http.NewRequest("POST", url, body)
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", writer.FormDataContentType())
    
    client := &http.Client{}
    resp, _ := client.Do(req)
    defer resp.Body.Close()
}
```
