## Initiate Request

<CodeGroup>

```bash cURL
curl --location --request POST 'https://gptproto.com/api/v3/openai/sora-2-pro/image-to-video' \
--header 'Authorization: GPTPROTO_API_KEY' \
--header 'Content-Type: application/json' \
--data-raw '{
  "prompt": "Put a hat on the cat",
  "image": "https://tos.gptproto.com/resource/cat.png",
  "duration": 4
}'
```

```python Python
import requests
import json

url = "https://gptproto.com/api/v3/openai/sora-2-pro/image-to-video"

headers = {
    "Authorization": "GPTPROTO_API_KEY",
    "Content-Type": "application/json"
}

payload = {
    "prompt": "Put a hat on the cat",
    "image": "https://tos.gptproto.com/resource/cat.png",
    "duration": 4
}

response = requests.post(url, headers=headers, data=json.dumps(payload))
print(response.json())
```

```javascript JavaScript
const axios = require('axios');

const url = 'https://gptproto.com/api/v3/openai/sora-2-pro/image-to-video';

const headers = {
    'Authorization': 'GPTPROTO_API_KEY',
    'Content-Type': 'application/json'
};

    prompt: 'Put a hat on the cat',
    image: 'https://tos.gptproto.com/resource/cat.png',

    .then(response => console.log(response.data))
    .catch(error => console.error(error));
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/openai/sora-2-pro/image-to-video"
    
    payload := map[string]interface{}{
        "prompt":   "Put a hat on the cat",
        "image":    "https://tos.gptproto.com/resource/cat.png",
        "duration": 4,
    }
    
    jsonData, _ := json.Marshal(payload)
    
    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")
    
    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()
    
    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    fmt.Println(result)
}
```
