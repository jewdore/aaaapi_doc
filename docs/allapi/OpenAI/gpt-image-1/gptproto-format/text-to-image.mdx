## Initiate Request

<CodeGroup>

```bash cURL
curl --location --request POST "https://gptproto.com/api/v3/openai/gpt-image-1/text-to-image" \
--header "Authorization: GPTPROTO_API_KEY" \
--header "Content-Type: application/json" \
--data-raw '{
    "prompt": "Girl holding cat",
    "quality": "medium",
    "size": "1024x1024",
    "background": "auto",
    "enable_sync_mode": false,
    "response_format": "url"
}'
```

```python Python
import requests
import json

url = "https://gptproto.com/api/v3/openai/gpt-image-1/text-to-image"

headers = {
    "Authorization": "GPTPROTO_API_KEY",
    "Content-Type": "application/json"
}

payload = {
    "prompt": "Girl holding cat",
    "quality": "medium",
    "size": "1024x1024",
    "background": "auto",
    "enable_sync_mode": False,
    "response_format": "url"
}

response = requests.post(url, headers=headers, data=json.dumps(payload))
print(response.json())
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/openai/gpt-image-1/text-to-image"

    payload := map[string]interface{}{
        "prompt":           "Girl holding cat",
        "quality":          "medium",
        "size":             "1024x1024",
        "background":       "auto",
        "enable_sync_mode": false,
        "response_format":  "url",
    }

    jsonData, _ := json.Marshal(payload)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        fmt.Println(err)
        return
    }
    defer resp.Body.Close()

    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    fmt.Println(result)
}
```
