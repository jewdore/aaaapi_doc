## Initiate Request

```bash cURL
curl --request POST 'https://gptproto.com/api/v3/doubao/seedance-1-5-pro-251215/image-to-video' \
--header 'Authorization: GPTPROTO_API_KEY' \
--header 'Content-Type: application/json' \
--data-raw '{
    "image_url": "https://example.com/image.jpg",
    "prompt": "A beautiful scene with smooth motion"
}'
```

```python Python
import requests
import json

url = "https://gptproto.com/api/v3/doubao/seedance-1-5-pro-251215/image-to-video"

headers = {
    "Authorization": "GPTPROTO_API_KEY",
    "Content-Type": "application/json"
}

payload = {
    "image_url": "https://example.com/image.jpg",
    "prompt": "A beautiful scene with smooth motion"
}

response = requests.post(url, headers=headers, data=json.dumps(payload))
print(response.json())
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/doubao/seedance-1-5-pro-251215/image-to-video"
    
    payload := map[string]interface{}{
        "image_url": "https://example.com/image.jpg",
        "prompt": "A beautiful scene with smooth motion",
    }
    
    jsonData, _ := json.Marshal(payload)
    
    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")
    
    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()
}
```
