```python Python
import requests
import json

url = "https://gptproto.com/api/v3/doubao/seedance-1-5-pro-251215/image-to-video"

headers = {
    "Authorization": "GPTPROTO_API_KEY",
    "Content-Type": "application/json"

    "image": "https://example.com/input-image.jpg",
    "prompt": "A beautiful sunset scene with waves crashing on the beach"
```

```javascript JavaScript
const fetch = require('node-fetch');

    'Authorization': 'GPTPROTO_API_KEY',
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/doubao/seedance-1-5-pro-251215/image-to-video"
    
    payload := map[string]interface{}{
        "image": "https://example.com/input-image.jpg",
        "prompt": "A beautiful sunset scene with waves crashing on the beach",
    }
    
    jsonPayload, _ := json.Marshal(payload)
    
    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonPayload))
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")
    
    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()
    
    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
