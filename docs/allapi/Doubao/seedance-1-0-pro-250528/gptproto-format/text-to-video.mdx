```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/bytedance/doubao-seedance-1-0-pro-250528/text-to-video"

    payload := map[string]interface{}{
        "prompt":         "Multiple shots. A detective enters a dimly lit room. He examines the clues on the table and picks up an item. The camera shifts to him deep in thought.",
        "duration":       5,
        "aspect_ratio":   "1:1",
        "camera_fixed":   false,
        "resolution":     "480p",
        "generate_audio": true,
        "seed":           -1,
    }

    jsonData, _ := json.Marshal(payload)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "GPTPROTO_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := io.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
