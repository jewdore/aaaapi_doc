## Overview

This endpoint provides submit task functionality.

## Authentication

This endpoint requires authentication using a Bearer token.

<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key in the format: `Bearer YOUR_API_KEY`
</ParamField>

## Request Body

<ParamField body="image" type="string"  default="The URL of the image to generate an image from.">
  Image parameter
</ParamField>

<ParamField body="enable_base64_output" type="boolean"  default="false">
  Enable Base64 Output parameter
</ParamField>

<ParamField body="enable_sync_mode" type="boolean"  default="false">
  Enable Sync Mode parameter
</ParamField>

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/image-background-remover"

    payload := []byte(`{
"image": "The URL of the image to generate an image from.",
"enable_base64_output": false,
"enable_sync_mode": false
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
