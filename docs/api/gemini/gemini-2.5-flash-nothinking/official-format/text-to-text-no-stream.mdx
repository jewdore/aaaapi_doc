## Overview

This endpoint generates content using Gemini models via the official Gemini API format. It supports text generation, image analysis, and multi-turn conversations.

## Authentication

This endpoint requires authentication using an API key.

<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key
</ParamField>

## Path Parameters

<ParamField path="model" type="string" required default="gemini-2.5-flash-nothinking">
  The model to use for content generation. Available models: `gemini-2.5-flash-nothinking`, etc.
</ParamField>

## Request Body Parameters

<ParamField body="contents" type="array" required>
  An array of content objects representing the conversation history.

  <Expandable title="Content Object Structure">
    <ParamField body="role" type="string" required>
      The role of the message sender. Typically `"user"` or `"model"`.
    </ParamField>

    <ParamField body="parts" type="array" required>
      An array of parts that make up the message content.

      <Expandable title="Parts Object Structure">
        <ParamField body="text" type="string">
          Text content of the message.
        </ParamField>

        <ParamField body="inlineData" type="object">
          Inline data for images or other media (base64 encoded).
        </ParamField>

        <ParamField body="functionResponse" type="object">
          Response from a function call.
        </ParamField>

        <ParamField body="fileData" type="object">
          Reference to uploaded file data.
        </ParamField>

        <ParamField body="videoMetadata" type="object">
          Metadata for video content.
        </ParamField>

        <ParamField body="functionCall" type="object">
          Function call request.
        </ParamField>

        <ParamField body="functionCalls" type="array">
          Multiple function calls.
        </ParamField>

        <ParamField body="thoughtSignature" type="string">
          Signature for thought content.
        </ParamField>

        <ParamField body="thought" type="string">
          Thought content for reasoning models.
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1beta/models/gemini-2.5-flash-nothinking:generateContent"

    payload := map[string]interface{}{
        "contents": []map[string]interface{}{
            {
                "role": "user",
                "parts": []map[string]interface{}{
                    {
                        "text": "hello",
                    },
                },
            },
        },
    }

    jsonData, _ := json.Marshal(payload)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "sk-***********")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
