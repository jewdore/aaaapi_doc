## Overview

This endpoint provides file analysis functionality (no stream).

## Authentication

This endpoint requires authentication using an API key.

<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key
</ParamField>

## Path Parameters

<ParamField path="model" type="string" required default="gemini-2.5-pro">
  The model to use for the file analysis. Available models: `gemini-2.5-pro`, etc.
</ParamField>


## Request Body Parameters

<ParamField body="contents" type="array" required>
  An array of content objects representing the conversation history and file data.

  <Expandable title="Content Object Structure">
    <ParamField body="role" type="string">
      The role of the message sender. Typically `"user"` or `"model"`. Optional for single-turn requests.
    </ParamField>

    <ParamField body="parts" type="array" required>
      An array of parts that make up the message content. Can include both text and file data.

      <Expandable title="Parts Object Structure">
        <ParamField body="text" type="string">
          Text content of the message or prompt for analyzing the file.
        </ParamField>

        <ParamField body="file_data" type="object">
          Reference to file data for analysis (PDF, images, etc.).

          <Expandable title="File Data Structure">
            <ParamField body="mime_type" type="string" required>
              The MIME type of the file (e.g., `"application/pdf"`, `"image/jpeg"`, `"image/png"`).
            </ParamField>

            <ParamField body="file_uri" type="string" required>
              The URI of the file to analyze. Can be a public URL.
            </ParamField>
          </Expandable>
        </ParamField>

      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1beta/models/gemini-2.5-pro:generateContent"

    payload := map[string]interface{}{
        "contents": []map[string]interface{}{
            {
                "role": "user",
                "parts": []map[string]interface{}{
                    {
                        "text": "帮我分析这份文件",
                    },
                    {
                        "file_data": map[string]string{
                            "mime_type": "application/pdf",
                            "file_uri":  "https://oss.heyoos.com/ai-draw/material/Essential-English.pdf",
                        },
                    },
                },
            },
        },
    }

    jsonData, _ := json.Marshal(payload)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "sk-***********")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
