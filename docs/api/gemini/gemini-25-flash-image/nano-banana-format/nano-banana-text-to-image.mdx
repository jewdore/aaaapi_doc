## Overview

This endpoint provides nano-banana-text-to-image functionality.

## Authentication

This endpoint requires authentication using a Bearer token.

<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key in the format: `Bearer YOUR_API_KEY`
</ParamField>

## Request Body

<ParamField body="prompt" type="string" required default="The cat is climbing on the tree">

<ParamField body="aspect_ratio" type="string"  default="3:2">

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "gptproto/api/v3/google/nano-banana/text-to-image"

    payload := []byte(`{
"prompt": "The cat is climbing on the tree",
"aspect_ratio": "3:2",
"output_format": "jpeg",
"enable_sync_mode": false,
"enable_base64_output": true
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
