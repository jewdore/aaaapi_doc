## Overview

This endpoint provides nano-banana-text-to-image functionality.

## Parameters

| Parameter | Type | Required | Default | Description |
|-----------|------|----------|---------|-------------|
| `prompt` | string | ✅ Yes | `The cat is climbing on the tree` | Prompt parameter |
| `aspect_ratio` | string | ❌ No | `3:2` | Aspect Ratio parameter |
| `output_format` | string | ❌ No | `jpeg` | Output Format parameter |
| `enable_sync_mode` | boolean | ❌ No | `false` | Enable Sync Mode parameter |
| `enable_base64_output` | boolean | ❌ No | `true` | Enable Base64 Output parameter |

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "gptproto/api/v3/google/nano-banana/text-to-image"

    payload := []byte(`{
"prompt": "The cat is climbing on the tree",
"aspect_ratio": "3:2",
"output_format": "jpeg",
"enable_sync_mode": false,
"enable_base64_output": true
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
