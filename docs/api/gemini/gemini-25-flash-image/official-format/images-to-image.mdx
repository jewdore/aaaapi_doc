```python Python
import requests
import json
import base64

url = "https://gptproto.com/v1beta/models/gemini-2.5-flash-image:generateContent"
headers = {
    "Authorization": "sk-***********",
    "Content-Type": "application/json"
}

# Read and encode multiple images
image_files = ["image1.jpg", "image2.jpg", "image3.png"]
image_parts = []

for img_file in image_files:
    with open(img_file, "rb") as image_file:
        image_data = base64.b64encode(image_file.read()).decode('utf-8')
        mime_type = "image/png" if img_file.endswith(".png") else "image/jpeg"
        image_parts.append({
            "inline_data": {
                "mime_type": mime_type,
                "data": image_data
            }
        })

data = {
    "contents": [
        {
            "role": "user",
            "parts": [
                {
                    "text": "Combine these images into a creative collage"
                }
            ] + image_parts
        }
    ],
    "generationConfig": {
        "responseModalities": [
            "TEXT",
            "IMAGE"
        ]
    }
}

response = requests.post(url, headers=headers, json=data)
result = response.json()
print(json.dumps(result, indent=2))
```

## Path Parameters


## Parameters

| Parameter | Type | Required | Default | Description |
|-----------|------|----------|---------|-------------|
| `contents` | array | ✅ Yes | - | An array of content objects representing the generation prompt and input images.   The role of the message sender. Typically `"user"`. Optional for single-turn requests. |
| `parts` | array | ✅ Yes | - | An array of parts that make up the message content, including text prompt and multiple input images.   Text prompt for image transformation. |
| `inline_data` | object | ❌ No | - | Input image data in base64 format. Can be specified multiple times for multiple images.   The MIME type of the image (e.g., `"image/jpeg"`, `"image/png"`). |
| `data` | string | ✅ Yes | - | Base64-encoded image data. |
| `generationConfig` | object | ❌ No | - | Configuration for the image generation behavior.   Response modalities. For image generation, use `["TEXT", "IMAGE"]`. |
| `imageConfig` | object | ❌ No | - | Image-specific configuration.   Aspect ratio for the generated image. Examples: `"16:9"`, `"1:1"`, `"9:16"`. |
