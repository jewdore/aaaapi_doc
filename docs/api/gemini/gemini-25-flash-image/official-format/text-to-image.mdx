## Overview

This endpoint generates images using Gemini Flash Image model via the official Gemini API format. It supports text-to-image generation with configurable image parameters.

## Authentication

This endpoint requires authentication using an API key.

<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key
</ParamField>

## Path Parameters

<ParamField path="model" type="string" required default="gemini-2.5-flash-image">
  The model to use for image generation. Available models: `gemini-2.5-flash-image`, etc.
</ParamField>

## Request Body Parameters

<ParamField body="contents" type="array" required>
  An array of content objects representing the generation prompt.

  <Expandable title="Content Object Structure">
    <ParamField body="role" type="string">
      The role of the message sender. Typically `"user"`. Optional for single-turn requests.
    </ParamField>

    <ParamField body="parts" type="array" required>
      An array of parts that make up the message content.

      <Expandable title="Parts Object Structure">
        <ParamField body="text" type="string">
          Text prompt for image generation.
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="generationConfig" type="object">
  Configuration for the image generation behavior.

  <Expandable title="Generation Config Options">
    <ParamField body="responseModalities" type="array">
      Response modalities. For image generation, use `["TEXT", "IMAGE"]`.
    </ParamField>

    <ParamField body="imageConfig" type="object">
      Image-specific configuration.

      <Expandable title="Image Config Options">
        <ParamField body="aspectRatio" type="string">
          Aspect ratio for the generated image. Examples: `"16:9"`, `"1:1"`, `"9:16"`.
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

package main

import (

            },
        },

    jsonData, _ := json.Marshal(payload)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))

    if err != nil {
        panic(err)

    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
