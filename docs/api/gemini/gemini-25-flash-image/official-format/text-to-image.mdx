## Supported inputs & outputs :

Inputs: Text, Images

Outputs: Text and image

## Overview

This endpoint provides text to image functionality.

## Authentication

This endpoint requires authentication using a Bearer token.

<ParamField header="Authorization" default="sk-***********" type="string" required>
  Your API key in the format: `YOUR_API_KEY`
</ParamField>

## Request Body

<ParamField body="contents" default='[{"parts": [{"text": "Hi, can you create a 3d rendered image of a pig with wings and a top hat flying over a happy futuristic scifi city with lots of greenery?"}]}]' type="array">
  Contents parameter
</ParamField>

<ParamField body="generationConfig" default='{"responseModalities": ["TEXT", "IMAGE"], "imageConfig": {"aspectRatio": "16:9"}}' type="object">
  Generationconfig parameter
</ParamField>

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "gptproto/v1beta/models/gemini-2.5-flash-image:generateContent"

    payload := []byte(`{
"contents": [
{
"parts": [
{
"text": "Hi, can you create a 3d rendered image of a pig with wings and a top hat flying over a happy futuristic scifi city with lots of greenery?"
}
]
}
],
"generationConfig": {
"responseModalities": [
"TEXT",
"IMAGE"
],
"imageConfig": {
"aspectRatio": "16:9"
}
}
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
