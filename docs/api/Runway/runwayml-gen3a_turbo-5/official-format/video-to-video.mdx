```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/runway/v1/pro/video2video"

    payload := []byte(`{
"video": "http://source.baidu.com/attachments/video/runway/20241117/28ed9d05e1dc84bede45541855500aa9.mp4",
"model": "runwayml-gen3a_turbo-5",
"prompt": "pixel style",
"options": {
"structure_transformation": 0.5,
"flip": true
}
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```

## Parameters

| Parameter | Type | Required | Default | Description |
|-----------|------|----------|---------|-------------|
| `video` | string | ❌ No | `http://source.baidu.com/attachments/video/runway/20241117/28ed9d05e1dc84bede45541855500aa9.mp4` | Video parameter |
| `model` | string | ✅ Yes | `runwayml-gen3a_turbo-5` | The model to use for the request |
| `prompt` | string | ✅ Yes | `pixel style` | Prompt parameter |
| `options` | object | ❌ No | `{&quot;structure_transformation&quot;: 0.5, &quot;flip&quot;: true}` | Options parameter |
