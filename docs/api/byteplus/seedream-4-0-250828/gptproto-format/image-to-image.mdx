## Overview

This endpoint provides image to image functionality.

## Parameters

| Parameter | Type | Required | Default | Description |
|-----------|------|----------|---------|-------------|
| `prompt` | string | ✅ Yes | `Generate a twin photo from two images and take a photo in the snow` | Image generation prompt describing the desired image content |
| `image` | array | ❌ No | `[&quot;https://heyoo.oss-ap-southeast-1.aliyuncs.com/ai-draw/java/test/cto.jpeg&quot;, &quot;https://heyoo.oss-ap-southeast-1.aliyuncs.com/ai-draw/java/test/cto.jpeg&quot;]` | Input image URL(s) for image-to-image generation |
| `model` | string | ✅ Yes | `seedream-4-0-250828` | The model to use for the request |
| `sequential_image_generation` | string | ❌ No | `disabled` | Sequential Image Generation parameter |
| `response_format` | string | ❌ No | `url` | Response format: "url" (image URL) or "b64_json" (base64 encoded) |
| `size` | string | ❌ No | `2K` | Output image resolution (e.g., 1024x1024, 2K, 4K) |
| `stream` | boolean | ❌ No | `false` | Whether to stream the response |
| `watermark` | boolean | ❌ No | `false` | Whether to add watermark to generated image |

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/images/generations"

    payload := []byte(`{
"prompt": "Generate a twin photo from two images and take a photo in the snow",
"image": [
"https://heyoo.oss-ap-southeast-1.aliyuncs.com/ai-draw/java/test/cto.jpeg",
"https://heyoo.oss-ap-southeast-1.aliyuncs.com/ai-draw/java/test/cto.jpeg"
],
"model": "seedream-4-0-250828",
"sequential_image_generation": "disabled",
"response_format": "url",
"size": "2K",
"stream": false,
"watermark": false
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
