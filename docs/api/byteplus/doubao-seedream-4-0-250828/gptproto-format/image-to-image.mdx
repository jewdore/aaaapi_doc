## Overview

This endpoint provides image to image functionality.

## Parameters

| Parameter | Type | Required | Default | Description |
|-----------|------|----------|---------|-------------|
| `image` | array | ❌ No | `[&quot;https://heyoo.oss-ap-southeast-1.aliyuncs.com/ai-draw/java/test/cto.jpeg&quot;, &quot;https://heyoo.oss-ap-southeast-1.aliyuncs.com/ai-draw/java/test/cto.jpeg&quot;]` | Input image URL(s) for image-to-image generation |
| `hd` | boolean | ❌ No | `true` | Hd parameter |
| `model` | string | ✅ Yes | `doubao-seedream-4-0-250828` | The model to use for the request |
| `n` | integer | ❌ No | `1` | Number of completions to generate |
| `prompt` | string | ✅ Yes | `Generate close-up photos of characters in the snow` | Image generation prompt describing the desired image content |
| `prompt_enhancement` | boolean | ❌ No | `false` | Prompt Enhancement parameter |
| `quality` | string | ❌ No | `auto` | Quality parameter |
| `reflection_enabled` | boolean | ❌ No | `false` | Reflection Enabled parameter |
| `response_format` | string | ❌ No | `url` | Response format: "url" (image URL) or "b64_json" (base64 encoded) |
| `seed` | integer | ❌ No | `12` | Random seed for reproducible results |
| `size` | string | ❌ No | `1024x1280` | Output image resolution (e.g., 1024x1024, 2K, 4K) |

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1/images/generations"

    payload := []byte(`{
"image": [
"https://heyoo.oss-ap-southeast-1.aliyuncs.com/ai-draw/java/test/cto.jpeg",
"https://heyoo.oss-ap-southeast-1.aliyuncs.com/ai-draw/java/test/cto.jpeg"
],
"hd": true,
"model": "doubao-seedream-4-0-250828",
"n": 1,
"prompt": "Generate close-up photos of characters in the snow",
"prompt_enhancement": false,
"quality": "auto",
"reflection_enabled": false,
"response_format": "url",
"seed": 12,
"size": "1024x1280"
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
