## Overview

This endpoint provides image to video functionality.

## Parameters

| Parameter | Type | Required | Default | Description |
|-----------|------|----------|---------|-------------|
| `prompt` | string | ✅ Yes | `flow` | Prompt parameter |
| `negative_prompt` | string | ❌ No | - | Negative Prompt parameter |
| `image` | string | ❌ No | `https://img.jpg` | Image parameter |
| `image_tail` | string | ❌ No | - | Image Tail parameter |
| `aspect_ratio` | string | ❌ No | `1:1` | Aspect Ratio parameter |
| `mode` | string | ❌ No | `std` | Mode parameter |
| `duration` | string | ❌ No | `5` | Duration parameter |

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/kling/v1/videos/image2video"

    payload := []byte(`{
"prompt": "flow",
"negative_prompt": "",
"image": "https://img.jpg",
"image_tail": "",
"aspect_ratio": "1:1",
"mode": "std",
"duration": "5"
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
