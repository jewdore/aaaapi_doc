```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/bytedance/seedance-1-0-pro-250528/text-to-video"

    payload := []byte(`{
    "prompt": "The camera enters a bright and cozy band rehearsal room—empty of people but full of instruments: guitars, amps, bass, drums. The scene is rich in realistic detail. [Slow pan] as the drumhead subtly quivers in the silence.",
    "aspect_ratio": "16:9",
    "resolution": "480p",
    "duration": "5",
    "seed": -1
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("authorization", "sk-xx")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```

## Parameters

| Parameter | Type | Required | Default | Description |
|-----------|------|----------|---------|-------------|
| `prompt` | string | ✅ Yes | - | Video generation prompt describing the desired video content |
| `aspect_ratio` | string | ❌ No | `16:9` | Video aspect ratio (e.g., 16:9, 9:16, 1:1) |
| `resolution` | string | ❌ No | `480p` | Output video resolution (e.g., 480p, 720p, 1080p) |
| `duration` | string | ❌ No | `5` | Video duration in seconds |
| `seed` | integer | ❌ No | `-1` | Random seed for reproducible results (-1 for random) |
