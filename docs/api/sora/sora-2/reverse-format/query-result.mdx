## Overview

This endpoint provides query result functionality.

## Parameters

| Parameter | Type | Required | Default | Description |
|-----------|------|----------|---------|-------------|
| `data` | object | ❌ No | `{&quot;id&quot;: &quot;sora-2:task_xxxx&quot;, &quot;model&quot;: &quot;sora-2&quot;, &quot;outputs&quot;: [&quot;https://oss.heyoos.com/sora/xxxxxxxxx.mp4&quot;], &quot;urls&quot;: {&quot;get&quot;: &quot;https://gptproto.com/api/v3/predictions/sora-2:task_xxxx/result&quot;}, &quot;status&quot;: &quot;completed&quot;, &quot;error&quot;: null, &quot;executionTime&quot;: 0, &quot;timings&quot;: {&quot;inference&quot;: 0}, &quot;has_nsfw_contents&quot;: [], &quot;created_at&quot;: &quot;2025-10-16 21:06:38&quot;}` | Data parameter |
| `message` | string | ❌ No | `success` | Message parameter |
| `code` | integer | ❌ No | `200` | Code parameter |

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "gptproto/api/v3/predictions/sora-2:task_xxx/result"

    payload := []byte(`{
"data": {
"id": "sora-2:task_xxxx",
"model": "sora-2",
"outputs": [
"https://oss.heyoos.com/sora/xxxxxxxxx.mp4"
],
"urls": {
"get": "https://gptproto.com/api/v3/predictions/sora-2:task_xxxx/result"
},
"status": "completed",
"error": null,
"executionTime": 0,
"timings": {
"inference": 0
},
"has_nsfw_contents": [],
"created_at": "2025-10-16 21:06:38"
},
"message": "success",
"code": 200
}`)

    req, _ := http.NewRequest("GET", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
