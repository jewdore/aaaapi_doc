## Overview

This endpoint provides image to image functionality.

## Authentication

This endpoint requires authentication using a Bearer token.

<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key in the format: `YOUR_API_KEY`
</ParamField>

## Request Body

<ParamField body="image" type="file" required>
  image parameter
</ParamField>

<ParamField body="prompt" type="string" required>
  prompt parameter
</ParamField>

<ParamField body="mask" type="file" required>
  mask parameter
</ParamField>

<ParamField body="model" type="string" required default="gpt-4o-image-vip">
  model parameter
</ParamField>

  n parameter

## Request Example

<CodeGroup>

```bash cURL
curl -X POST "https://gptproto.com/v1/images/edits" \
  -H "Authorization: YOUR_API_KEY" \
  -F "image=@/path/to/your/image.png" \
  -F "prompt=Transform this image into a watercolor painting" \
  -F "mask=@/path/to/your/mask.png" \
  -F "model=gpt-4o-image-vip" \
  -F "n=1"
```

```python Python
import requests

url = "https://gptproto.com/v1/images/edits"
headers = {
    "Authorization": "YOUR_API_KEY"
}

files = {
    'image': open('/path/to/your/image.png', 'rb'),
    'mask': open('/path/to/your/mask.png', 'rb')
}

data = {
    'prompt': 'Transform this image into a watercolor painting',
    'model': 'gpt-4o-image-vip',
    'n': '1'
}

response = requests.post(url, headers=headers, files=files, data=data)
result = response.json()
print(result)
```

```javascript JavaScript
const FormData = require('form-data');
const fs = require('fs');
const axios = require('axios');

const url = "https://gptproto.com/v1/images/edits";
const formData = new FormData();

formData.append('image', fs.createReadStream('/path/to/your/image.png'));
formData.append('mask', fs.createReadStream('/path/to/your/mask.png'));
formData.append('prompt', 'Transform this image into a watercolor painting');
formData.append('model', 'gpt-4o-image-vip');
formData.append('n', '1');

axios.post(url, formData, {
  headers: {
    'Authorization': 'YOUR_API_KEY',
    ...formData.getHeaders()
  }
})
.then(response => console.log(response.data))
.catch(error => console.error('Error:', error));
```

```go Go
package main

import (
    "bytes"
    "fmt"
    "io"
    "mime/multipart"
    "net/http"
    "os"
)

func main() {
    url := "https://gptproto.com/v1/images/edits"

    // Create multipart form
    body := &bytes.Buffer{}
    writer := multipart.NewWriter(body)

    // Add image file
    imageFile, _ := os.Open("/path/to/your/image.png")
    defer imageFile.Close()
    imagePart, _ := writer.CreateFormFile("image", "image.png")
    io.Copy(imagePart, imageFile)

    // Add mask file
    maskFile, _ := os.Open("/path/to/your/mask.png")
    defer maskFile.Close()
    maskPart, _ := writer.CreateFormFile("mask", "mask.png")
    io.Copy(maskPart, maskFile)

    // Add other fields
    writer.WriteField("prompt", "Transform this image into a watercolor painting")
    writer.WriteField("model", "gpt-4o-image-vip")
    writer.WriteField("n", "1")
    writer.Close()

    req, _ := http.NewRequest("POST", url, body)
    req.Header.Set("Authorization", "YOUR_API_KEY")
    req.Header.Set("Content-Type", writer.FormDataContentType())

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    responseBody, _ := io.ReadAll(resp.Body)
    fmt.Println(string(responseBody))
}
```
