## Overview

This endpoint provides text to video functionality.

## Authentication

This endpoint requires authentication using a Bearer token.

<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key in the format: `Bearer YOUR_API_KEY`
</ParamField>

## Request Body

<ParamField body="model" type="string" required default="sora-2">
  The video generation model to use. Defaults to sora-2.
</ParamField>

<ParamField body="prompt" type="string" required default="The character walks slowly and then sits on the ground">
  Text prompt that describes the video to generate.
</ParamField>

<ParamField body="size" type="string" required default="720x1280">
  Output resolution formatted as width x height. 720x1280 & 1280x720 ，Defaults to 720x1280.
</ParamField>

<ParamField body="seconds" type="string" default="4">
  Clip duration in seconds.4, 8, 12 . Defaults to 4 seconds.
</ParamField>

<CodeGroup>
```bash cURL
curl -X POST "https://gptproto.com/v1/videos" \
--header 'Authorization: sk-xxx' \
--form 'model="sora-2"' \
--form 'prompt="The character walks slowly and then sits on the ground"' \
--form 'size="720x1280"' \
--form 'seconds="4"'
```

```python Python
import http.client

API_KEY = "sk-xxx"

boundary = 'wL36Yn8afVp8Ag7AmP8qZ0SA4n1v9T'
body = b'\r\n'.join([
    f'--{boundary}'.encode(),
    b'Content-Disposition: form-data; name="prompt"',
    b'',
    b'She turns around and smiles, then slowly walks out of the frame.',
    f'--{boundary}'.encode(),
    b'Content-Disposition: form-data; name="model"',
    b'',
    b'sora-2',
    f'--{boundary}'.encode(),
    b'Content-Disposition: form-data; name="size"',
    b'',
    b'1280x720',
    f'--{boundary}'.encode(),
    b'Content-Disposition: form-data; name="seconds"',
    b'',
    b'4',
    f'--{boundary}--'.encode(),
])

conn = http.client.HTTPSConnection("gptproto.com")
conn.request("POST", "/v1/videos", body, {
    'Authorization': API_KEY,
    'Content-Type': f'multipart/form-data; boundary={boundary}'
})
res = conn.getresponse()
print(res.read().decode("utf-8"))
conn.close()
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1/videos"

    payload := []byte(`{
"prompt": "A video of a horse dancing.",
"size": "720*1280",
"duration": 4
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
