## Overview

This endpoint provides submit task functionality.

## Authentication

This endpoint requires authentication using a Bearer token.


<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key in the format: `YOUR_API_KEY`
</ParamField>


## Request Parameter
<ParamField path="model" type="string" required default="sora-2">
  The video generation model to use. 
</ParamField>

<ParamField path="scene" type="string" required default="text-to-video">
  The scene to use.
</ParamField>

## Request Body


<ParamField body="orientation" type="string"  default="portrait">
  portrait 、landscape
</ParamField>

<ParamField body="prompt" type="string" required default="She turns around and smiles, then slowly walks out of the frame.">
  The positive prompt for the generation.
</ParamField>


<ParamField body="duration" type="integer"  default="4">
  The duration of the generated video in seconds.4, 8, 12
</ParamField>

<ParamField body="character_url" type="string" default="https://oss.heyoos.com/2025/11/11/fffe18ab440d4fadb16b4563c4cd18d1.mp4">
  Non-human IP video URL.
</ParamField>

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/api/v3/openai/reverse/sora-2/text-to-video"

    payload := []byte(`{
"orientation": "portrait",
"prompt": "She turns around and smiles, then slowly walks out of the frame.",
"duration": 4,
"character_url": "https://oss.heyoos.com/2025/11/11/fffe18ab440d4fadb16b4563c4cd18d1.mp4"
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
