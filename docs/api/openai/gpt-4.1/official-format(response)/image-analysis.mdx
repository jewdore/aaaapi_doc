## Overview

This endpoint provides image-to-text functionality using vision-enabled models in response mode.  provide image URLs to extract text content, describe scenes, or analyze visual information.

## Parameters

| Parameter | Type | Required | Description |
|-----------|------|----------|-------------|
| **Authorization** (Header) | string | Yes | Your API key in the format: `sk-*****` |
| **Content-Type** (Header) | string | Yes | Must be set to `application/json` |
| **model** | string | Yes | The model to use for the request. Must be a vision-enabled model. |
| **input** | array | Yes | Array of message objects with role and content |
| **input[].role** | string | Yes | Role of the message sender (`user` or `assistant`) |
| **input[].content** | array | Yes | Array of content objects |
| **input[].content[].type** | string | Yes | Type of content (`input_text` or `input_image`) |
| **input[].content[].text** | string | Conditional | Text prompt (required when type is `input_text`) |
| **input[].content[].image_url** | string | Conditional | Image URL or base64 encoded image (required when type is `input_image`) |
| **stream** | boolean | No | Whether to stream the response |

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1/responses"

    payload := map[string]interface{}{
        "model": "gpt-4.1",
        "input": []map[string]interface{}{
            {
                "role": "user",
                "content": []map[string]interface{}{
                    {
                        "type": "input_text",
                        "text": "What is in this image?",
                    },
                    {
                        "type":     "input_image",
                        "image_url": "data:image/jpeg;base64,${base64Image}",
                    },
                },
            },
        },
        "stream": false,
    }

    jsonData, _ := json.Marshal(payload)
    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "sk-*****")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
