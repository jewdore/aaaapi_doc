## Overview

This endpoint provides image-to-text functionality using vision-enabled models in response mode.  provide image URLs to extract text content, describe scenes, or analyze visual information.

## Authentication

This endpoint requires authentication using a Bearer token.

<ParamField header="Authorization" default="sk-***********" type="string" required>
  Your API key in the format: `sk-*****`
</ParamField>

## Request Body

<ParamField body="model" default="o4-mini-deep-research" type="string" required>
  The model to use for the request. Must be a vision-enabled model.
</ParamField>

<ParamField body="input" type="array" required default='[{"role": "user", "content": [{"type": "input_text", "text": "What is in this image?"}, {"type": "input_image", "image_url": "data:image/jpeg;base64,${base64Image}"}]}]'>
  Array of message objects with role and content. Each message contains:
  - `role`: "user" or "assistant"
  - `content`: Array of content objects supporting the following types:
    - **input_text**: Text prompt with `text` field
    - **input_image**: Image input with `image_url` field, supports:
      - Base64 encoded images: `data:image/jpeg;base64,{base64_string}`
      - Supported formats: JPG, PNG

  Example with base64 encoded image:
  ```json
  [
    {
      "role": "user",
      "content": [
        {
          "type": "input_text",
          "text": "What is in this image?"
        },
        {
          "type": "input_image",
          "image_url": "data:image/jpeg;base64,${base64Image}"
        }
      ]
    }
  ]
  ```
</ParamField>


<ParamField body="stream" default={false} type="boolean">
  Whether to stream the response
</ParamField>

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1/responses"

    payload := map[string]interface{}{
        "model": "o4-mini-deep-research",
        "input": []map[string]interface{}{
            {
                "role": "user",
                "content": []map[string]interface{}{
                    {
                        "type": "input_text",
                        "text": "What is in this image?",
                    },
                    {
                        "type":     "input_image",
                        "image_url": "data:image/jpeg;base64,${base64Image}",
                    },
                },
            },
        },
    }

    jsonData, _ := json.Marshal(payload)
    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "sk-*****")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```

## Response

<ResponseField name="id" type="string">
  Unique identifier for the response
</ResponseField>

<ResponseField name="object" type="string">
  Object type, always "response"
</ResponseField>

<ResponseField name="created" type="integer">
  Unix timestamp of when the response was created
</ResponseField>

<ResponseField name="model" type="string">
  The model used for generating the response
</ResponseField>

<ResponseField name="output" type="string">
  The generated text output (the extracted text or image analysis)
</ResponseField>

<ResponseField name="usage" type="object">
  Token usage statistics

  <Expandable title="Usage Object">
    <ResponseField name="prompt_tokens" type="integer">
      Number of tokens in the input
    </ResponseField>

    <ResponseField name="completion_tokens" type="integer">
      Number of tokens in the output
    </ResponseField>

    <ResponseField name="total_tokens" type="integer">
      Total number of tokens used
    </ResponseField>
  </Expandable>
</ResponseField>
