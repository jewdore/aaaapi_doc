## Overview

Retrieve the status and results of music generation tasks submitted to [Suno AI](https://suno.com). This endpoint allows you to monitor the progress of asynchronous music generation tasks and retrieve the generated audio URL once complete.

Related Guides: [Suno API Documentation](https://suno.com/docs/api) | [Task Management Guide](https://suno.com/guides/task-management)

After submitting a music generation request via the [Text to Audio](/api/suno/official-format/text-to-audio) endpoint, use this endpoint to check the task status and retrieve the generated audio file.

```python Python
import requests

task_id = "15410652-357c-47fa-8ad9-a4a145fbfa14"
url = f"https://gptproto.com/v1/suno/fetch/{task_id}"

headers = {
    "Authorization": "YOUR_API_KEY",
    "Content-Type": "application/json"
}

response = requests.get(url, headers=headers)
result = response.json()

if result["code"] == "success" and result["data"]:
    task = result["data"][0]
    print(f"Status: {task['status']}")

    if task['status'] == 'completed':
        print(f"Audio URL: {task['audio_url']}")
```

```go Go
package main

import (
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

type TaskData struct {
    ID       string `json:"id"`
    Status   string `json:"status"`
    AudioURL string `json:"audio_url,omitempty"`
    VideoURL string `json:"video_url,omitempty"`
    Title    string `json:"title,omitempty"`
}

type Response struct {
    Code    string      `json:"code"`
    Data    []TaskData  `json:"data"`
    Message string      `json:"message"`
}

func main() {
    taskID := "15410652-357c-47fa-8ad9-a4a145fbfa14"
    url := fmt.Sprintf("https://gptproto.com/v1/suno/fetch/%s", taskID)

    req, _ := http.NewRequest("GET", url, nil)
    req.Header.Set("Authorization", "YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)

    var result Response
    json.Unmarshal(body, &result)

    if result.Code == "success" && len(result.Data) > 0 {
        task := result.Data[0]
        fmt.Printf("Status: %s\n", task.Status)

        if task.Status == "completed" {
            fmt.Printf("Audio URL: %s\n", task.AudioURL)
        }
    }
}
```

## Task Status Values

| Status | Description | Next Action |
|--------|-------------|-------------|
| `queued` | Task is waiting in queue to be processed | Continue polling |
| `processing` | Task is currently being processed | Continue polling, check progress |
| `completed` | Task completed successfully, audio is ready | Download audio from `audio_url` |
| `failed` | Task failed | Check `error_message`, retry if needed |
